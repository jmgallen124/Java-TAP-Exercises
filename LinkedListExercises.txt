// Import main package
package LinkedListExercises;

// Import functions
import java.util.*;

// Main class
public class LinkedListExercise1 {

    // Main method
    public static void main(String[] args) {
        
        // Declare LinkedList
        LinkedList<String> games = new LinkedList<String>();
        
        // Append items to LinkedList
        games.add("Muramasa Rebirth");
        games.add("Dust: An Elysian Tail");
        games.add("Tails of Iron");
        games.add("F.I.S.T.: Forged in Shadow Torch");
        games.add("Devil Survivor 2: Record Breaker");
        
        // Output LinkedList
        System.out.println(games);
        
        // Append item to end of LinkedList
        games.addLast("Timespinner");
        
        // Output LinkedList again
        System.out.println(games);        
        
    }
    
}

------
